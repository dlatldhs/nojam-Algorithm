# 정렬
### 선택정렬 알고리즘
```cpp
#include <bits/stdc++.h>
using namespace std;
vector<int> v;
int n,m,min_value,min_index,temp,k,i;

void select_sort() {
    
	for ( i = 0 ; i < (v.size()-1) ; i++ ) {
		min_value = 10000;
		min_index = k;
		for ( k = i ; k < (v.size()) ; k++ ) {
			temp = v[i];
			if ( v[k] < min_value ) {
				// result.push_back(v[k]);
				min_index = k;
				min_value = v[k];
			}
		}
		
		v[i] = min_value;
		v[min_index] = temp;
		// cout << v[i];
	}
}

int main() {
	cin.clear();
	cin >> n;
	for ( int i =0 ; i < n ; i++ ) {
		cin >> m;
		v.push_back(m);
	}
	 
	select_sort();
	
	for ( int i=0; i< v.size() ; i++ ) {
		if ( i==0 ) {
			cout << v[i];
		} else {
			cout << " " << v[i];
		}
	}
	
	return 0;
}
```

### 삽입정렬 알고리즘
```cpp
#include <bits/stdc++.h>
using namespace std;
vector<int> v;
int n,m,min_value,min_index,temp,k,i,key;

void insert_sort() {
	for ( int i=1; i < v.size() ; i++ ) {
		
		key = v[i];
		int k=i-1; // 밑에 for 문 에서 v[k]>key 라는 조건이 충족하지 않으면 k 의 초기값이 초기화 되지 않아서 에러 발생함 그래서 여기서 초기화를 한번 시켜주는 거임
		for (; k>= 0 && v[k]>key; k-- ) {
			// cout << v[k];
			v[k+1] = v[k];
		}
		v[k+1] = key;
	}
}

int main() {
	cin.clear();
	cin >> n;
	for ( int i =0 ; i < n ; i++ ) {
		cin >> m;
		v.push_back(m);
	}
	 
	insert_sort();
	
	for ( int i=0; i< v.size() ; i++ ) {
		if ( i==0 ) {
			cout << v[i];
		} else {
			cout << " " << v[i];
		}
	}
	
	return 0;
}
```
