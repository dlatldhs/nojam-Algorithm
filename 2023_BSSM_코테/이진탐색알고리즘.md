```cpp
    cin >> ary[i];
  }
  endd = n - 1;

  while (endd - start >= 0) {
    mid = (endd + start) / 2;
    if (ary[mid] == m) {
      cout << mid + 1;
      return 0;
    } else if (ary[mid] > m) {
      endd = mid - 1;
    } else {
      start = mid + 1;
    }
  }

  cout << "원소가 존재하지 않습니다.";
  return 0;
}
```

## 실습1
```cpp
#include <bits/stdc++.h>
using namespace std;
int n, m; // n 개수 , m 찾는값
int ary[104];
int start, endd, mid,cnt;
int main() {
  ios_base::sync_with_stdio(false); cin.tie(NULL); cout.tie(NULL);
  cin >> n >> m;
  for (int i = 0; i < n; i++) {
    cin >> ary[i];
  }
  endd = n - 1;

  while (endd - start >= 0) {
    mid = (endd + start) / 2;
    if (ary[mid] == m) {
			break;
		} else if (ary[mid] > m) {
      endd = mid - 1;
    } else {
      start = mid + 1;
    }
  }
	
	int l_idx = mid -1 , r_idx = mid +1;
			while ( l_idx >= 0 && ary[l_idx] == m ) {
				cnt++;
				l_idx--;
			}
			while ( r_idx < n && ary[r_idx] == m ) {
				cnt++;
				r_idx++;
			}
	
  cout << cnt+1;
  return 0;
}
```
